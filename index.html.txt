  <!-- ================= LOGIC & INTERACTION SCRIPTS ================= -->
  <script>
    // ================= PROGRESS & CARD SWITCHER LOGIC =================
    let currentStepIndex = 1;
    const totalSteps = 5;

    // Set initial progress bar value
    function updateProgress(step) {
      const progressBar = document.getElementById('progress-bar');
      const percentage = (step / totalSteps) * 100;
      gsap.to(progressBar, { width: `${percentage}%`, duration: 0.6, ease: "power2.out" });
    }
    updateProgress(currentStepIndex);

    // Initial stagger animation on page load for Step 1 components
    window.addEventListener('DOMContentLoaded', () => {
      const step1elements = document.querySelectorAll('#step-1 .animate-me');
      gsap.fromTo(step1elements, 
        { opacity: 0, y: 30 },
        { opacity: 1, y: 0, duration: 0.8, stagger: 0.2, ease: "power2.out" }
      );
    });

    // Handle seamless staggered navigation transitions
    function goToNextStep() {
      if (currentStepIndex >= totalSteps) return;

      const currentStepEl = document.getElementById(`step-${currentStepIndex}`);
      const nextStepIndex = currentStepIndex + 1;
      const nextStepEl = document.getElementById(`step-${nextStepIndex}`);

      // Transition the active step out smoothly
      gsap.to(currentStepEl, {
        opacity: 0,
        y: -30,
        scale: 0.95,
        duration: 0.5,
        ease: "power2.inOut",
        onComplete: () => {
          currentStepEl.classList.add('hidden');
          nextStepEl.classList.remove('hidden');

          // Match the elements to run staggered reveal animations
          const elementsToAnimate = nextStepEl.querySelectorAll('.animate-me');
          gsap.set(nextStepEl, { opacity: 1, y: 0, scale: 1 }); // Main container maintains relative position
          gsap.set(elementsToAnimate, { opacity: 0, y: 30 });

          gsap.to(elementsToAnimate, {
            opacity: 1,
            y: 0,
            duration: 0.6,
            stagger: 0.15,
            ease: "power2.out"
          });

          // Polaroid specific check on step 4 to initialize hover parallax
          if (nextStepIndex === 4) {
            initPolaroidParallax();
          }

          currentStepIndex = nextStepIndex;
          updateProgress(currentStepIndex);
        }
      });
    }

    // ================= STEP 4 POLAROID INTERACTION =================
    function initPolaroidParallax() {
      const polaroid = document.getElementById('polaroid');
      if (!polaroid) return;

      polaroid.addEventListener('mousemove', (e) => {
        const rect = polaroid.getBoundingClientRect();
        const x = e.clientX - rect.left - rect.width / 2;
        const y = e.clientY - rect.top - rect.height / 2;
        
        // Map movements to visual tilt boundaries
        const rotX = -(y / rect.height) * 22; // limits to safe rotations
        const rotY = (x / rect.width) * 22;

        gsap.to(polaroid, {
          rotateX: rotX,
          rotateY: rotY,
          transformPerspective: 800,
          ease: "power2.out",
          duration: 0.3
        });
      });

      polaroid.addEventListener('mouseleave', () => {
        gsap.to(polaroid, {
          rotateX: 0,
          rotateY: 0,
          rotation: -3, // returns back to natural aesthetic layout rotation
          duration: 0.6,
          ease: "power3.out"
        });
      });
    }

    // ================= CELEBRATION CLIMAX SEQUENCE =================
    let isCelebrating = false; // Linked to Three.js environment

    function triggerCelebrate() {
      const btn = document.getElementById('btn-celebrate');
      const finalMsg = document.getElementById('final-message');

      if (!btn || !finalMsg) return;

      // 1. Disable and fade out the celebrate button action
      btn.disabled = true;
      gsap.to(btn, { 
        opacity: 0, 
        scale: 0.9, 
        duration: 0.4, 
        onComplete: () => {
          btn.classList.add('hidden');
          
          // 2. Elegantly slide and fade up the special final message
          finalMsg.classList.remove('hidden');
          gsap.fromTo(finalMsg, 
            { opacity: 0, y: 25 },
            { opacity: 1, y: 0, duration: 1.2, ease: "power3.out" }
          );
        } 
      });

      // 3. Set celebratory state for 3D Background Hearts logic
      isCelebrating = true;

      // 4. Trigger intense corner bursts of canvas confetti immediately
      confetti({
        particleCount: 100,
        angle: 60,
        spread: 60,
        origin: { x: 0, y: 1 },
        colors: ['#f43f5e', '#ec4899', '#f43f5e', '#fbbf24', '#22d3ee']
      });
      confetti({
        particleCount: 100,
        angle: 120,
        spread: 60,
        origin: { x: 1, y: 1 },
        colors: ['#f43f5e', '#ec4899', '#f43f5e', '#fbbf24', '#22d3ee']
      });

      // 5. Shower continuously with standard particles and themed emojis for 5 seconds
      const duration = 5 * 1000;
      const animationEnd = Date.now() + duration;
      const defaults = { startVelocity: 30, spread: 360, ticks: 60, zIndex: 100 };

      const randomInRange = (min, max) => Math.random() * (max - min) + min;

      const interval = setInterval(() => {
        const timeLeft = animationEnd - Date.now();

        if (timeLeft <= 0) {
          return clearInterval(interval);
        }

        const particleCount = 25 * (timeLeft / duration);

        // Standard colored sparkles showering down
        confetti(Object.assign({}, defaults, {
          particleCount,
          origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 }
        }));
        confetti(Object.assign({}, defaults, {
          particleCount,
          origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 }
        }));

        // Beautiful floating emoji confetti
        confetti({
          ...defaults,
          particleCount: 6,
          flat: true,
          scalar: 2.2,
          shapes: ['text'],
          shapeOptions: {
            text: {
              value: ['??', '??', '?', '??']
            }
          },
          origin: { x: randomInRange(0.2, 0.8), y: Math.random() - 0.2 }
        });
      }, 250);
    }

    // ================= LAYER 2: THREE.JS 3D BACKGROUND HEARTS =================
    const canvas = document.getElementById('three-canvas');
    if (canvas) {
      const scene = new THREE.Scene();
      const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 1000);
      camera.position.z = 8;

      const renderer = new THREE.WebGLRenderer({
        canvas: canvas,
        alpha: true,
        antialias: true
      });
      renderer.setSize(window.innerWidth, window.innerHeight);
      renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));

      // Handle viewport dimensions changes gracefully
      window.addEventListener('resize', () => {
        camera.aspect = window.innerWidth / window.innerHeight;
        camera.updateProjectionMatrix();
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
      });

      // Ambient and soft light points to define clean glossy details
      const ambientLight = new THREE.AmbientLight(0xffffff, 0.65);
      scene.add(ambientLight);

      const dirLight = new THREE.DirectionalLight(0xffe3ec, 1.4);
      dirLight.position.set(5, 10, 7);
      scene.add(dirLight);

      const pointLight = new THREE.PointLight(0xff0055, 1.2, 10);
      pointLight.position.set(-5, -5, 2);
      scene.add(pointLight);

      // Create a smooth 3D heart ExtrudeGeometry
      const heartShape = new THREE.Shape();
      heartShape.moveTo(25, 25);
      heartShape.bezierCurveTo(25, 25, 20, 0, 0, 0);
      heartShape.bezierCurveTo(-30, 0, -30, 35, -30, 35);
      heartShape.bezierCurveTo(-30, 55, -10, 77, 25, 95);
      heartShape.bezierCurveTo(60, 77, 80, 55, 80, 35);
      heartShape.bezierCurveTo(80, 35, 80, 0, 50, 0);
      heartShape.bezierCurveTo(35, 0, 25, 25, 25, 25);

      const extrudeSettings = { depth: 10, bevelEnabled: true, bevelSegments: 3, steps: 1, bevelSize: 4, bevelThickness: 4 };
      const geometry = new THREE.ExtrudeGeometry(heartShape, extrudeSettings);
      geometry.center();
      geometry.scale(0.012, 0.012, 0.012); // Balance object dimensions

      // Instantiate hearts matching responsive counts
      const hearts = [];
      const isMobile = window.innerWidth < 768;
      const heartCount = isMobile ? 12 : 24; 
      const colors = [0xf43f5e, 0xec4899, 0xd946ef, 0xff85a1, 0xffb3c6];

      for (let i = 0; i < heartCount; i++) {
        const color = colors[Math.floor(Math.random() * colors.length)];
        const material = new THREE.MeshPhysicalMaterial({
          color: color,
          roughness: 0.15,
          metalness: 0.1,
          clearcoat: 1.0,
          clearcoatRoughness: 0.1,
          sheen: new THREE.Color(0xff85a1)
        });

        const mesh = new THREE.Mesh(geometry, material);
        mesh.position.set(
          (Math.random() - 0.5) * 12,
          (Math.random() - 0.5) * 10,
          (Math.random() - 0.5) * 5
        );
        mesh.rotation.set(Math.random() * Math.PI, Math.random() * Math.PI, Math.random() * Math.PI);

        const scale = 0.4 + Math.random() * 0.7;
        mesh.scale.set(scale, scale, scale);

        scene.add(mesh);

        hearts.push({
          mesh: mesh,
          baseX: mesh.position.x,
          baseY: mesh.position.y,
          baseZ: mesh.position.z,
          rotSpeedX: (Math.random() - 0.5) * 0.01,
          rotSpeedY: (Math.random() - 0.5) * 0.015,
          rotSpeedZ: (Math.random() - 0.5) * 0.005,
          driftSpeed: 0.4 + Math.random() * 1.2,
          phase: Math.random() * Math.PI * 2
        });
      }

      // Main Loop variables
      const clock = new THREE.Clock();

      function animate() {
        requestAnimationFrame(animate);
        const time = clock.getElapsedTime();

        hearts.forEach(heart => {
          if (!isCelebrating) {
            // Floating State: Gentle drifting & breathing waves
            heart.mesh.rotation.x += heart.rotSpeedX;
            heart.mesh.rotation.y += heart.rotSpeedY;
            heart.mesh.rotation.z += heart.rotSpeedZ;

            heart.mesh.position.y = heart.baseY + Math.sin(time * 0.4 + heart.phase) * 0.35;
            heart.mesh.position.x = heart.baseX + Math.cos(time * 0.3 + heart.phase) * 0.25;
          } else {
            // Climax State: Gracefully swirl outward, float upwards, and fade away
            heart.mesh.rotation.x += heart.rotSpeedX * 5;
            heart.mesh.rotation.y += heart.rotSpeedY * 5;
            heart.mesh.position.y += 0.12; 
            heart.mesh.position.x += Math.sin(time * 6 + heart.phase) * 0.08 + Math.sign(heart.baseX) * 0.06;

            if (heart.mesh.scale.x > 0.01) {
              heart.mesh.scale.multiplyScalar(0.975);
            }
          }
        });

        renderer.render(scene, camera);
      }
      animate();
    }
  </script>
